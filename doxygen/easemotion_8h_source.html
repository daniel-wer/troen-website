367  {
368  switch (_behaviour)

461  virtual void getValueInNormalizedRange(float t, value_type& result) const
462  {
463  if (_motions.empty())

465  result = 0;
466  osg::notify(osg::WARN) << "CompositeMotion::getValueInNormalizedRange no Motion in the CompositeMotion, add motion to have result" << std::endl;
467  return;
468  }
469  for (MotionList::const_iterator it = _motions.begin(); it != _motions.end(); ++it)
470  {

472  float durationInRange = motion->getDuration() / getDuration();
473  if (t < durationInRange)
474  {
475  float tInRange = t/durationInRange * motion->getDuration();
476  motion->getValueAt( tInRange, result);
477  return;
478  } else
479  t = t - durationInRange;
480  }
481  osg::notify(osg::WARN) << "CompositeMotion::getValueInNormalizedRange did find the value in range, something wrong" << std::endl;
482  result = 0;
483  }
